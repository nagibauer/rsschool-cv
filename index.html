    x = rand(0..100)
    puts "\nЯ загадал число от 1 до 100. Угадай, какое?"
    puts "У тебя есть 10 попыток"

    times = 10

    1.upto(times) do |pp|

        print "\nПопытка #{pp}: "
        answ = gets.strip.to_i

        if answ < x
            puts "Нет, больше!\n"
        elsif answ > x
            puts "Нет, меньше!\n"
        elsif answ == x 
            puts "Браво!"
            exit
        end

        if pp == times && answ != x
            puts "Так и не угадал"
            puts "Правильный ответ: #{x}"
        end

        print "Осталось попыток #{times - pp}\n"
    end
